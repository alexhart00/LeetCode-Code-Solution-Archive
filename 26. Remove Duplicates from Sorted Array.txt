class Solution {
    public int removeDuplicates(int[] nums) {
        HashSet<Integer> uniqueInetegers = new HashSet<>();
        int j = 0;
        for(int i = 0; i < nums.length; i++){
            if(uniqueInetegers.add(nums[i])){
                nums[j] = nums[i];
                j++;
            }
        }
        return uniqueInetegers.size();
    }
}
