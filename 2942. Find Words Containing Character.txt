class Solution {
    public List<Integer> findWordsContaining(String[] words, char x) {
        List<Integer> returnArray = new ArrayList<>();
        for(int i = 0; i < words.length; i++)
            for(char c : words[i].toCharArray())
                if(c == x){
                    returnArray.add(i);
                    break;
                }

        return returnArray;
    }
}
