class Solution {
    public int rearrangeCharacters(String s, String target) {
        Map<Character, Integer> sCharacterOccurrences = new HashMap<>();
        Map<Character, Integer> targetCharacterOccurrences = new HashMap<>();
        int max = Integer.MAX_VALUE;

        for(char c : s.toCharArray())
            sCharacterOccurrences.put(c, sCharacterOccurrences.getOrDefault(c, 0) + 1);

        for(char c : target.toCharArray())            
            targetCharacterOccurrences.put(c, targetCharacterOccurrences.getOrDefault(c, 0) + 1);

        for(Map.Entry<Character, Integer> entry : targetCharacterOccurrences.entrySet()) {
            char c = entry.getKey();
            int targetCount = entry.getValue();
            int sourceCount = sCharacterOccurrences.getOrDefault(c, 0);
            max = Math.min(sourceCount / targetCount, max);
        }

        return max;
    }
}
