class Solution {
    public ListString generatePossibleNextMoves(String currentState) {
        ListString returnList = new ArrayList();
        StringBuilder sb = new StringBuilder(currentState);
        for(int i = 0; i  sb.length() - 1; i++){
            if(sb.charAt(i) == '+' && sb.charAt(i+1) == '+')
                returnList.add(sb.replace(i, i+2, --).toString());
            sb.replace(0, sb.length(), currentState);
        }
        return returnList;
    }
}
