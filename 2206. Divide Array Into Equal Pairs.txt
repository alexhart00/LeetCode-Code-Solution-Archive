class Solution {
    public boolean divideArray(int[] nums) {
        Map<Integer, Integer> numOccurrences = new HashMap<>();
        for(int num : nums)
            numOccurrences.put(num, numOccurrences.getOrDefault(num, 0) + 1);
        
        for(int count : numOccurrences.values())
            if(count % 2 != 0)
                return false;
        
        return true;
    }
}
