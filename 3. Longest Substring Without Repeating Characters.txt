class Solution {
    public int lengthOfLongestSubstring(String s) {
        Set<Character> charSet = new HashSet<>();
        int maxLength = 0;
        for (int i = 0; i < s.length(); i++) {
            charSet.clear(); 
            for (int j = i; j < s.length(); j++) {
                if (!charSet.add(s.charAt(j))) { 
                    break;
                }
                maxLength = Math.max(maxLength, j - i + 1);
            }
        }
        return maxLength;
    }
}
