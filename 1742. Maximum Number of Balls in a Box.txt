class Solution {
    public int countBalls(int lowLimit, int highLimit) {
        HashMap<Integer, Integer> numberboxCount = new HashMap<>();
        int max = 0;
        
        for(int i = lowLimit; i <= highLimit; i++){
            int boxNumber = boxNumber(i);
            numberboxCount.put(boxNumber, numberboxCount.getOrDefault(boxNumber, 0) + 1);
            max = Math.max(max, numberboxCount.get(boxNumber));
        }
            
        return max;
    }

    private int boxNumber(int num){
        int sum = 0;

        while(num > 0){
            sum += num % 10;
            num /= 10;
        }
        
        return sum;
    }
}
