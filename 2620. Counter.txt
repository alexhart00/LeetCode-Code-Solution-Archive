/**
 * @param {number} n
 * @return {Function} counter
 */
var createCounter = function(n) {
    let returnValue = n - 1
    return function(n) {
        returnValue += 1;
        return returnValue;
    };
};

/** 
 * const counter = createCounter(10)
 * counter() // 10
 * counter() // 11
 * counter() // 12
 */
