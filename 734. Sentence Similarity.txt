class Solution {
    public boolean areSentencesSimilar(String[] sentence1, String[] sentence2, List<List<String>> similarPairs) {
        if(sentence1.length != sentence2.length)
            return false;

        HashMap<String, Set<String>> pairMap = new HashMap<>();
        for(List<String> pair : similarPairs){
            pairMap.computeIfAbsent(pair.get(0), k -> new HashSet<>()).add(pair.get(1));
            pairMap.computeIfAbsent(pair.get(1), k -> new HashSet<>()).add(pair.get(0));
        }

        for(int i = 0; i < sentence1.length; i++){
            if(sentence1[i].equals(sentence2[i]))
                continue;
            if (!pairMap.containsKey(sentence1[i]) || !pairMap.get(sentence1[i]).contains(sentence2[i]))
                return false;
        }

        return true;
    }
}
