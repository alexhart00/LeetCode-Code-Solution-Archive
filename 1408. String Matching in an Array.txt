class Solution {
    public List<String> stringMatching(String[] words) {
        List<String> returnArray = new ArrayList<>();
        
        for(String subStringWord : words){
            for(String word : words){
                if(word.contains(subStringWord) && !subStringWord.equals(word)){
                    returnArray.add(subStringWord);
                    break;
                }
            }
        }

        return returnArray;
    }
}
