class Solution {
    public boolean checkRecord(String s) {
        int totalA = 0;

        if(s.contains("LLL"))
            return false;

        for(int i = 0; i < s.length(); i++)
        {
            if(s.charAt(i) == 'A'){
                totalA++;
                if (totalA == 2) 
                    return false; 
            }
        }

        return true;
    }
}
