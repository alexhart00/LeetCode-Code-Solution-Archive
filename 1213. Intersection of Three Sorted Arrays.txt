class Solution {
    public List<Integer> arraysIntersection(int[] arr1, int[] arr2, int[] arr3) {
        Map<Integer, Integer> numOccurrences = new HashMap<>();
        List<Integer> returnList = new ArrayList<>();
        
        countOccurrences(arr1, numOccurrences);
        countOccurrences(arr2, numOccurrences);
        countOccurrences(arr3, numOccurrences);

        for(Map.Entry<Integer, Integer> map : numOccurrences.entrySet())
            if(map.getValue() >= 3)
                returnList.add(map.getKey());
                
        Collections.sort(returnList);
        return returnList;
    }

    private void countOccurrences(int[] arr, Map<Integer, Integer> map) {
        for (int num : arr)
            map.put(num, map.getOrDefault(num, 0) + 1);
    }
}
