class Solution {
    char[] digitsCharacters;
    List<String> returnList = new ArrayList<>();
    Map<Integer, List<Character>> phoneNumberMap = Map.of(
            2, List.of('a', 'b', 'c'),
            3, List.of('d', 'e', 'f'),
            4, List.of('g', 'h', 'i'),
            5, List.of('j', 'k', 'l'),
            6, List.of('m', 'n', 'o'),
            7, List.of('p', 'q', 'r', 's'),
            8, List.of('t', 'u', 'v'),
            9, List.of('w', 'x', 'y', 'z')
        );
    
    public List<String> letterCombinations(String digits) {
        if(digits.length() != 0){
            digitsCharacters = digits.toCharArray();    
            recursiveNumberLetter(0, new StringBuilder());
        }
        return returnList;
    }

    private void recursiveNumberLetter(int currentLength, StringBuilder letterCombination){
        if(currentLength == digitsCharacters.length){
            returnList.add(letterCombination.toString());
            return;
        }
        
        for(char characterToCycle : phoneNumberMap.get(Character.getNumericValue(digitsCharacters[currentLength]))){
            recursiveNumberLetter(currentLength + 1,  new StringBuilder(letterCombination.toString()).append(characterToCycle));
        }
        return;
    }
}
